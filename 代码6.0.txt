<script>
    document.addEventListener('DOMContentLoaded', function() {
        const aiInput = document.getElementById('aiInput');
        const aiReply = document.getElementById('aiReply');
        const aiReplyText = document.getElementById('aiReplyText');
        const aiVoiceBtn = document.getElementById('aiVoiceBtn');
        const aiSubmitBtn = document.getElementById('aiSubmitBtn');

        // 首次聚焦清空默认提示文字
        let isFirstFocus = true;
        aiInput.addEventListener('focus', function() {
            if (isFirstFocus && this.innerText.includes("请输入您的诉求")) {
                this.innerHTML = '';
                isFirstFocus = false;
            }
        });

        // 只有有效输入失去焦点后，才弹出AI回复
        aiInput.addEventListener('blur', function() {
            const inputText = this.innerText.trim();
            // 过滤空内容、默认提示文字
            if (inputText && !inputText.includes("请输入您的诉求") && inputText.length >= 5) {
                aiReplyText.innerText = '已为您识别到诉求内容，点击下方按钮即可一键提交至诉求页面';
                aiReply.style.display = 'block';
            }
        });

        // 语音输入逻辑
        aiVoiceBtn.addEventListener('click', async function() {
            const btn = this;
            startSpeechRecognition(async (status, text) => {
                if (status === "onstart") {
                    btn.style.background = '#ef4444';
                    btn.innerHTML = '<i class="fa-solid fa-microphone-lines-slash"></i> 正在聆听...';
                } else if (status === "onend") {
                    const result = await stopSpeechRecognition();
                    btn.style.background = '#1a56db';
                    btn.innerHTML = '<i class="fa-solid fa-microphone"></i> 语音输入诉求';

                    if (result) {
                        aiInput.innerHTML = result;
                        isFirstFocus = false;
                        // 语音输入完成后，自动触发AI回复
                        aiReplyText.innerText = '已为您识别到诉求内容，点击下方按钮即可一键提交';
                        aiReply.style.display = 'block';
                    }
                } else if (status === "onerror") {
                    btn.style.background = '#1a56db';
                    btn.innerHTML = '<i class="fa-solid fa-microphone"></i> 语音输入诉求';
                }
            });
        });

        // 提交逻辑
        aiSubmitBtn.addEventListener('click', function() {
            let inputText = aiInput.innerText.trim();
            if (inputText.includes("请输入您的诉求")) inputText = "";
            if (!inputText || inputText.length < 5) {
                alert("请先输入有效的诉求内容");
                return;
            }
            localStorage.setItem('ai_input_appeal', inputText);
            window.location.href = 'appeal-submit.html';
        });
    });
</script>

    // 修改诉求保存
    function updateAppeal(cardIndex, appealId) {
        // 打印日志，方便调试
        console.log("当前卡片下标：", cardIndex);
        console.log("当前诉求ID：", appealId);

        // 读取所有诉求数据
        const allAppeals = JSON.parse(localStorage.getItem('zhizhitong_appeals')) || [];
        // 查找诉求，两边都转字符串，解决类型不匹配
        const appealIndex = allAppeals.findIndex(a => String(a.id) === String(appealId));

        // 找不到诉求的兜底提示
        if (appealIndex === -1) {
            alert(`错误：未找到ID为${appealId}的诉求，请刷新页面后重试！`);
            return;
        }

        // 获取表单所有值
        const newStatus = document.getElementById(`statusSelect-${cardIndex}`).value;
        const newResult = document.getElementById(`resultText-${cardIndex}`).value;
        // 分派信息
        const handlerDept = document.getElementById(`handlerDept-${cardIndex}`).value;
        const handlerPerson = document.getElementById(`handlerPerson-${cardIndex}`).value;
        const dispatchRemark = document.getElementById(`dispatchRemark-${cardIndex}`).value;
        // 办结信息
        const finishDesc = document.getElementById(`finishDesc-${cardIndex}`).value;

        // 管理员绝不能设置为已完成
        if (newStatus === '已完成') {
            alert('违规操作！已完成状态只能由居民核验后自动生成，管理员不可手动设置！');
            return;
        }

        // 实时读取最新上传的图片
        const realtimeAppeals = JSON.parse(localStorage.getItem('zhizhitong_appeals')) || [];
        const realtimeIndex = realtimeAppeals.findIndex(a => String(a.id) === String(appealId));
        const realFinishImages = realtimeAppeals[realtimeIndex]?.finishImages || [];

        // ========== 核心校验规则 ==========
        // 1. 提交待核验，必须填办结说明+上传办结凭证

            const appealIndex = allAppeals.findIndex(a => String(a.id) === String(appealId));
            if (appealIndex === -1) {

                return;
            }

            // 初始化图片数组
            if (!allAppeals[appealIndex].finishImages) {
                allAppeals[appealIndex].finishImages = [];
            }
            let currentImages = allAppeals[appealIndex].finishImages;

            // 限制最多3张
            if (currentImages.length + files.length > 3) {
                alert('最多只能上传3张办结凭证图片');
                e.target.value = '';
                return;
            }

            // 遍历上传图片
            for (let i = 0; i < files.length; i++) {
                const file = files[i];
                // 格式校验
                if (!file.type.match('image/jpeg|image/png')) {
                    alert(`第${i+1}张图片格式错误，仅支持JPG/PNG`);
                    continue;
                }
                // 大小校验
                if (file.size > 5 * 1024 * 1024) {
                    alert(`第${i+1}张图片超过5MB，请压缩后上传`);
                    continue;
                }

                // 读取图片并实时预览+保存
                const reader = new FileReader();
                reader.onload = function(event) {
                    const base64 = event.target.result;
                    currentImages.push(base64);

                    // 保存到本地存储
                    allAppeals[appealIndex].finishImages = currentImages;
                    localStorage.setItem('zhizhitong_appeals', JSON.stringify(allAppeals));

                    // 实时渲染预览（选完立刻显示图片）
                    previewContainer.innerHTML = currentImages.map((img, idx) => `
                    <div class="appeal-image-item" style="width: 60px; height: 60px; border-radius: 8px; overflow: hidden;">
                        <img src="${img}" alt="办结凭证${idx+1}" style="width:100%;height:100%;object-fit:cover;">
                    </div>
                `).join('');
                };
                reader.readAsDataURL(file);
            }

            // 清空文件框，允许重复上传
            setTimeout(() => {
                e.target.value = '';
            }, 500);
        }
    });

    // 加载用户诉求数据
    function loadUserAppeals() {
        const currentUser = localStorage.getItem('current_user');
        const allAppeals = JSON.parse(localStorage.getItem('zhizhitong_appeals')) || [];
        // 筛选当前用户的诉求
        allUserAppeals = allAppeals.filter(appeal => appeal.phone === currentUser);

        // 默认渲染全部
        renderAppealList('all');
    }

    // 封装成独立的渲染列表函数，筛选时调用
    function renderAppealList(filterStatus = 'all') {
        const emptyState = document.getElementById('emptyState');
        const appealList = document.getElementById('appealList');

        // 按状态筛选
        let filteredList = allUserAppeals;
        if (filterStatus !== 'all') {
            filteredList = allUserAppeals.filter(item => item.status === filterStatus);
        }

        // 无诉求：显示空状态
        if (filteredList.length === 0) {
            emptyState.style.display = 'block';
            appealList.style.display = 'none';
            return;
        }

        // 有诉求：显示列表
        emptyState.style.display = 'none';
        appealList.style.display = 'flex';
        // 按提交时间倒序排列
        filteredList.sort((a, b) => new Date(b.submitTime) - new Date(a.submitTime));

        // 生成诉求卡片
        appealList.innerHTML = '';
        filteredList.forEach((appeal, index) => {
            const appealCard = document.createElement('div');
            // 根据状态设置卡片类名
            let statusClass = '';
            let statusText = '';
            let statusTagClass = '';
            switch(appeal.status) {
                case '待处理':
                    statusClass = 'pending';
                    statusText = '待受理';
                    statusTagClass = 'status-pending';
                    break;
                case '处理中':
                    statusClass = 'processing';
                    statusText = '处理中';
                    statusTagClass = 'status-processing';
                    break;
                case '待核验':
                    statusClass = 'verifying';
                    statusText = '待核验';
                    statusTagClass = 'status-verifying';
                    break;
                case '已完成':
                    statusClass = 'completed';
                    statusText = '已完成';
                    statusTagClass = 'status-completed';
                    break;
                default:
                    statusClass = 'pending';
                    statusText = '待受理';
                    statusTagClass = 'status-pending';
            }
            // 生成图片预览HTML
            let imagesHtml = '';
            if (appeal.images && appeal.images.length > 0) {
                imagesHtml = `<div class="appeal-images">`;
                appeal.images.forEach((img, imgIndex) => {
                    imagesHtml += `
                        <div class="appeal-image-item" onclick="openImageModal('${img}')">
                            <img src="${img}" alt="诉求图片${imgIndex+1}">
                        </div>
                    `;
                });
                imagesHtml += `</div>`;
            }

            // 生成处理结果+办结凭证HTML
            let resultHtml = '';
            if ((appeal.status === '已完成' || appeal.status === '待核验') && (appeal.result || appeal.finishDesc)) {
                // 办结凭证图片HTML
                let finishImagesHtml = '';
                if (appeal.finishImages && appeal.finishImages.length > 0) {
                    finishImagesHtml = `<div class="appeal-images" style="margin-top: 12px;">`;
                    appeal.finishImages.forEach((img, imgIndex) => {
                        finishImagesHtml += `
                <div class="appeal-image-item" onclick="openImageModal('${img}')">
                    <img src="${img}" alt="办结凭证${imgIndex+1}">
                </div>
            `;
                    });
                    finishImagesHtml += `</div>`;
                }

                resultHtml = `
        <div class="appeal-result show">
            <div class="result-title">
                <i class="fa-solid fa-check-circle"></i> 处置结果说明
            </div>
            <div class="result-content">${appeal.finishDesc || appeal.result}</div>
            ${finishImagesHtml}
        </div>
    `;
            }


            // 生成进度条HTML
            let progressHtml = `
    <div class="progress-box" style="margin-top: 12px;">
        <div class="progress-step" style="display: flex; justify-content: space-between; margin-bottom: 8px; font-size: 12px; color: #6b7280;">
            <span class="${appeal.status === '待处理' ? 'active' : ''}">待受理</span>
            <span class="${['已分配','处理中','待核验','已完成'].includes(appeal.status) ? 'active' : ''}">已分配</span>
            <span class="${['处理中','待核验','已完成'].includes(appeal.status) ? 'active' : ''}">处理中</span>
            <span class="${['待核验','已完成'].includes(appeal.status) ? 'active' : ''}">待核验</span>
            <span class="${appeal.status === '已完成' ? 'active' : ''}">已完成</span>
        </div>
        <div class="progress-bar" style="width: 100%; height: 6px; background: #e5e7eb; border-radius: 3px; overflow: hidden;">
            <div class="progress-fill" style="height: 100%; background: linear-gradient(90deg, #1a56db 0%, #3b82f6 100%); width: ${getProgressPercent(appeal.status)}; transition: all 0.3s;"></div>
        </div>
    </div>
`;

            // 生成按钮文本（初始默认）
            let btnText = '查看完整';
            // 给每个描述框和按钮加唯一标识（避免冲突）
            const descId = `appealDesc-${index}`;
            const btnId = `appealBtn-${index}`;
            // 诉求卡片HTML
            appealCard.className = `appeal-card ${statusClass}`;
            appealCard.innerHTML = `
                <div class="appeal-header">
                    <div class="appeal-type">
                        <i class="fa-solid fa-tag"></i> ${appeal.type}
                    </div>
                    <div class="appeal-status ${statusTagClass}">
                        ${statusText}
                    </div>
                </div>
                <div class="appeal-content">
                    <div class="appeal-desc" id="${descId}" onclick="toggleDescExpand('${descId}', '${btnId}')">
                        ${appeal.desc}
                    </div>
                    ${imagesHtml}
                </div>
                ${resultHtml}
                ${progressHtml}
                <div class="appeal-footer">
                    <div class="appeal-meta">
                        <div class="meta-item">
                            <i class="fa-solid fa-calendar"></i>
                            <span>提交时间：${appeal.submitTime}</span>
                        </div>
                        <div class="meta-item">
                            <i class="fa-solid fa-id-card"></i>
                            <span>诉求编号：${appeal.code || '无'}</span>
                        </div>
                        <div class="meta-item">
                            <i class="fa-solid fa-id-card"></i>
                            <span>联系方式：${appeal.phone.slice(0,3)}****${appeal.phone.slice(7)}</span>
                        </div>
                    </div>
                    <div class="appeal-actions">
                     <button class="action-btn btn-view" id="${btnId}" onclick="toggleDescExpand('${descId}', '${btnId}')">
                     <i class="fa-solid fa-eye"></i> ${btnText}
                     </button>
                     <!-- 新增：待核验状态的核验操作按钮 -->
                    ${appeal.status === '待核验' ? `
                    <button class="action-btn btn-verify-pass" onclick="handleVerifyPass('${appeal.id}')">
                    <i class="fa-solid fa-check"></i> 核验通过
                    </button>
                    <button class="action-btn btn-verify-reject" onclick="openVerifyModal('${appeal.id}')">
                    <i class="fa-solid fa-times"></i> 核验不通过
                    </button>
` : ''}
    <!-- 新增：已完成状态的评价按钮 -->
                    ${appeal.status === '已完成' && !appeal.score ? `
                    <button class="action-btn" style="background: #8b5cf6; color: white;" onclick="openScoreModal('${appeal.id}')">
                    <i class="fa-solid fa-star"></i> 去评价
                    </button>
` : ''}
                    </div>
                </div>
            `;
            appealList.appendChild(appealCard);
            // 判断内容是否超过2行，不够则隐藏按钮
            setTimeout(() => {
                const descElement = document.getElementById(descId);
                const btnElement = document.getElementById(btnId);
                // 计算2行的高度（行高 × 2）
                const lineHeight = parseFloat(window.getComputedStyle(descElement).lineHeight);

                // 若内容高度 ≤ 2行高度，隐藏按钮
                if (descElement.scrollHeight <= twoLineHeight) {
                    btnElement.classList.add('hidden');
                }
            }, 100);
        });
    }

    // 进度百分比计算函数
    window.getProgressPercent = function(status) {
        const statusMap = {
            "待处理": "20%",
            "已分配": "40%",
            "处理中": "60%",
            "待核验": "80%",
            "已完成": "100%"
        };
        return statusMap[status] || "20%";
    };

    // 打开核验驳回弹窗
    function openVerifyModal(appealId) {
        document.getElementById('currentVerifyAppealId').value = appealId;
        document.getElementById('verifyRemark').value = '';
        document.getElementById('verifyModal').classList.add('show');

    // 核验通过
    function handleVerifyPass(appealId) {
        if (!confirm('确认核验通过？该操作将把诉求状态更新为已完成，完成全流程闭环。')) {
            return;
        }
        const allAppeals = JSON.parse(localStorage.getItem('zhizhitong_appeals')) || [];
        const appealIndex = allAppeals.findIndex(a => String(a.id) === String(appealId));
        if (appealIndex === -1) {
            alert('诉求不存在');
            return;
        }
        // 更新状态
        allAppeals[appealIndex].status = '已完成';
        allAppeals[appealIndex].verifyResult = '通过';
        allAppeals[appealIndex].verifyTime = new Date().toLocaleString();
        allAppeals[appealIndex].verifyRemark = '居民核验通过，诉求办结';
        // 保存数据
        localStorage.setItem('zhizhitong_appeals', JSON.stringify(allAppeals));
        // 给管理员发送通知
        sendNotice('admin', '诉求核验通过', `诉求编号${allAppeals[appealIndex].code}已被居民核验通过，诉求正式办结。`);
        // 提示+刷新页面
        alert('核验通过！诉求已正式办结，您可以对本次服务进行评价。');
        loadUserAppeals();
    }
    // 核验不通过（驳回）
    function handleVerifyReject() {
        const appealId = document.getElementById('currentVerifyAppealId').value;
        const verifyRemark = document.getElementById('verifyRemark').value.trim();
        if (!verifyRemark || verifyRemark.length < 5) {
            alert('请填写驳回原因（至少5字）');
            return;
        }
        const allAppeals = JSON.parse(localStorage.getItem('zhizhitong_appeals')) || [];
        const appealIndex = allAppeals.findIndex(a => String(a.id) === String(appealId));
        if (appealIndex === -1) {
            alert('诉求不存在');
            return;
        }
        // 更新状态：回退到处理中
        allAppeals[appealIndex].status = '处理中';
        allAppeals[appealIndex].verifyResult = '驳回';
        allAppeals[appealIndex].verifyTime = new Date().toLocaleString();
        allAppeals[appealIndex].verifyRemark = verifyRemark;
        // 保存数据
        localStorage.setItem('zhizhitong_appeals', JSON.stringify(allAppeals));
        // 给管理员发送通知
        sendNotice('admin', '诉求被驳回', `诉求编号${allAppeals[appealIndex].code}被居民驳回，驳回原因：${verifyRemark}，请重新处置。`);
        // 关闭弹窗+提示+刷新
        closeVerifyModal();
        alert('驳回已提交！管理员将重新处置您的诉求，后续进展会及时同步。');
        loadUserAppeals();
    }

    }

    // 提交评价
    function handleScoreSubmit() {
        const appealId = document.getElementById('currentScoreAppealId').value;
        const score = document.getElementById('currentScore').value;
        const scoreRemark = document.getElementById('scoreRemark').value.trim();
        if (score == 0) {
            alert('请选择星级评价');
            return;
        }
        const allAppeals = JSON.parse(localStorage.getItem('zhizhitong_appeals')) || [];
        const appealIndex = allAppeals.findIndex(a => String(a.id) === String(appealId));
        if (appealIndex === -1) {
            alert('诉求不存在');
            return;
        }
        // 保存评价
        allAppeals[appealIndex].score = score;
        allAppeals[appealIndex].scoreRemark = scoreRemark;
        // 保存数据
        localStorage.setItem('zhizhitong_appeals', JSON.stringify(allAppeals));
        // 给管理员发送通知
        sendNotice('admin', '收到居民评价', `诉求编号${allAppeals[appealIndex].code}收到居民${score}星评价，评价内容：${scoreRemark || '无'}`);
        // 关闭弹窗+提示+刷新
        closeScoreModal();
        alert('评价提交成功！感谢您的反馈，我们将持续优化服务。');
        loadUserAppeals();
    }

    // 图片上传预览
    const fileInput = document.getElementById('fileInput');
    const previewContainer = document.getElementById('previewContainer');
    const uploadContainer = document.getElementById('uploadContainer');
    let uploadedFiles = []; // 存储上传的图片base64
    fileInput.addEventListener('change', function(e) {
        const files = e.target.files;
        if (files.length === 0) return;
        // 限制最多3张图片
        if (uploadedFiles.length + files.length > 3) {
            alert('最多只能上传3张图片');
            return;
        }
        for (let i = 0; i < files.length; i++) {
            const file = files[i];
            // 验证文件类型
            if (!file.type.match('image/jpeg|image/png')) {
                alert('请上传JPG或PNG格式的图片');
                continue;
            }
            // 验证文件大小
            if (file.size > 5 * 1024 * 1024) {
                alert('单张图片大小不能超过5MB');
                continue;
            }
            // 读取图片并预览
            const reader = new FileReader();
            reader.onload = function(event) {
                const base64 = event.target.result;
                uploadedFiles.push(base64);
                // 创建预览元素
                const previewItem = document.createElement('div');
                previewItem.className = 'preview-item';
                previewItem.innerHTML = `
                        <img src="${base64}" class="preview-img" alt="诉求图片">
                        <div class="preview-remove" onclick="removeImage(${uploadedFiles.length - 1})">×</div>
                    `;
                previewContainer.appendChild(previewItem);
            };
            reader.readAsDataURL(file);
        }
        // 清空文件输入，允许重复选择同一文件
        fileInput.value = '';
    });

    // 移除图片
    function removeImage(index) {
        uploadedFiles.splice(index, 1);
        previewContainer.innerHTML = '';
        // 重新渲染预览
        uploadedFiles.forEach((base64, i) => {
            const previewItem = document.createElement('div');
            previewItem.className = 'preview-item';
            previewItem.innerHTML = `
                    <img src="${base64}" class="preview-img" alt="诉求图片">
                    <div class="preview-remove" onclick="removeImage(${i})">×</div>
                `;
            previewContainer.appendChild(previewItem);
        });
    }

        if (!validateForm()) return;


        // 生成唯一诉求编号（每天从001开始）
        const now = new Date();

        const month = String(now.getMonth() + 1).padStart(2, '0');
        const day = String(now.getDate()).padStart(2, '0');

    // 1. AI智能分类，自动匹配诉求类型
    document.getElementById('autoClassifyBtn').addEventListener('click', async function() {
        const appealDesc = document.getElementById('appealDesc').value.trim();
        if (!appealDesc || appealDesc.length < 10) {
            alert('请先输入完整的诉求描述（至少10字）');
            document.getElementById('appealDesc').focus();
            return;
        }
        // 按钮加载状态
        this.disabled = true;
        this.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> AI正在分析...';
        // 模拟AI智能分类，自动选择诉求类型
        let type = '其他';
        const desc = appealDesc;
        if(desc.includes('电梯')||desc.includes('纠纷')||desc.includes('邻里')) type='纠纷调解';
        else if(desc.includes('灯')||desc.includes('坏')||desc.includes('维修')||desc.includes('设施')) type='设施报修';
        else if(desc.includes('老人')||desc.includes('志愿')) type='志愿服务';
        else if(desc.includes('政策')||desc.includes('社保')||desc.includes('养老')) type='政策咨询';
        else if(desc.includes('建议')||desc.includes('建设')) type='社区建议';

        document.getElementById('appealType').value = type;

        // 恢复按钮状态
        this.disabled = false;
        this.innerHTML = '<i class="fa-solid fa-robot"></i> AI智能分类';
    });

                    speakText("语音输入完成");

// ==================== 通用错误处理函数 ====================
// 清除所有错误提示
function clearErrors() {
    const errors = document.querySelectorAll('.error-message');
    errors.forEach(error => error.classList.remove('show'));
}

// 显示错误提示+输入框抖动动画
function showError(inputId, message) {
    const errorElement = document.getElementById(inputId + 'Error');
    if (errorElement) {
        errorElement.textContent = message;
        errorElement.classList.add('show');
    }
    // 抖动动画
    const input = document.getElementById(inputId);
    if (input) {
        input.style.animation = 'none';
        setTimeout(() => input.style.animation = 'shake 0.5s ease', 10);
        input.focus();
    }
    // 注入抖动样式
    if (!document.getElementById('shakeStyle')) {
        const style = document.createElement('style');
        style.id = 'shakeStyle';
        style.textContent = `
      @keyframes shake {
        0%, 100% { transform: translateX(0); }
        10%, 30%, 50%, 70%, 90% { transform: translateX(-5px); }
        20%, 40%, 60%, 80% { transform: translateX(5px); }
      }
    `;
        document.head.appendChild(style);
    }
}

// ==================== 验证码倒计时通用函数 ====================
function startCountdown(btnElement, totalSeconds = 60) {
    let countdown = totalSeconds;
    btnElement.disabled = true;
    btnElement.textContent = `重新获取(${countdown}s)`;
    const timer = setInterval(() => {
        countdown--;
        btnElement.textContent = `重新获取(${countdown}s)`;
        if (countdown <= 0) {
            clearInterval(timer);
            btnElement.disabled = false;
            btnElement.textContent = '获取验证码';
        }
    }, 1000);
}

// 登录状态展示与退出按钮通用逻辑
window.addEventListener('DOMContentLoaded', function() {
    const isLogin = localStorage.getItem('is_login') === 'true';
    const currentUser = localStorage.getItem('current_user');
    const isAdmin = localStorage.getItem('is_admin') === 'true';
    const pagePath = window.location.pathname;

    // 排除登录/注册/忘记密码页面，不执行导航控制
    const excludePages = ['login.html', 'register.html', 'forgot-password.html'];
    const isExcludePage = excludePages.some(page => pagePath.includes(page));

    if (!isExcludePage) {
        // 获取所有导航元素
        const loginNavItem = document.getElementById('loginNavItem');
        const profileNavItem = document.getElementById('profileNavItem');
        const adminNavItem = document.getElementById('adminNavItem');
        const noticeNavItem = document.getElementById('noticeNavItem');
        const logoutNavItem = document.getElementById('logoutNavItem');
        const logoutBtn = document.getElementById('logoutBtn');

        // 登录状态判断
        if (isLogin && currentUser) {
            // 管理员登录逻辑
            if (isAdmin) {
                loginNavItem && (loginNavItem.style.display = 'none');
                profileNavItem && (profileNavItem.style.display = 'none');
                adminNavItem && (adminNavItem.style.display = 'block');
                noticeNavItem && (noticeNavItem.style.display = 'none');
                logoutNavItem && (logoutNavItem.style.display = 'block');
            }
            // 普通用户登录逻辑
            else {
                loginNavItem && (loginNavItem.style.display = 'none');
                profileNavItem && (profileNavItem.style.display = 'block');
                adminNavItem && (adminNavItem.style.display = 'none');
                noticeNavItem && (noticeNavItem.style.display = 'block');
                logoutNavItem && (logoutNavItem.style.display = 'block');
                // 更新消息小红点
                updateNoticeBadge();
            }

            // 退出登录按钮统一逻辑
            if (logoutBtn) {
                logoutBtn.onclick = function() {
                    if (confirm('确定要退出登录吗？')) {
                        localStorage.removeItem('is_login');
                        localStorage.removeItem('current_user');
                        localStorage.removeItem('is_admin');
                        window.location.href = 'index.html';
                    }
                };
            }
        }
        // 未登录逻辑
        else {
            loginNavItem && (loginNavItem.style.display = 'block');
            profileNavItem && (profileNavItem.style.display = 'none');
            adminNavItem && (adminNavItem.style.display = 'none');
            noticeNavItem && (noticeNavItem.style.display = 'none');
            logoutNavItem && (logoutNavItem.style.display = 'none');
        }
    }
});

// 3. 讯飞星火AI语音识别核心（真方言识别，纯前端可运行）
// ==============================================
/**
 * 讯飞语音识别核心函数
 * @param {Blob} audioBlob 录音音频文件
 * @returns {Promise<string>} 识别后的文本
 */
async function xunfeiSpeechToText(audioBlob) {
    try {
        // 1. 鉴权签名生成
        const url = "wss://ws-api.xfyun.cn/v2/iat";
        const host = "ws-api.xfyun.cn";
        const date = new Date().toUTCString();
        const signatureOrigin = `host: ${host}\ndate: ${date}\nGET /v2/iat HTTP/1.1`;

        // HMAC-SHA256签名（纯前端实现）
        const encoder = new TextEncoder();
        const keyData = encoder.encode(AI_CONFIG.xunfei.apiSecret);
        const messageData = encoder.encode(signatureOrigin);
        const key = await crypto.subtle.importKey("raw", keyData, { name: "HMAC", hash: "SHA-256" }, false, ["sign"]);
        const signatureBuffer = await crypto.subtle.sign("HMAC", key, messageData);
        const signature = btoa(String.fromCharCode(...new Uint8Array(signatureBuffer)));

        // 2. 构建鉴权头
        const authorization = `api_key="${AI_CONFIG.xunfei.apiKey}", algorithm="hmac-sha256", headers="host date request-line", signature="${signature}"`;
        const wsUrl = `${url}?authorization=${btoa(authorization)}&date=${btoa(date)}&host=${btoa(host)}`;

        // 3. WebSocket连接讯飞接口
        return new Promise((resolve, reject) => {
            const ws = new WebSocket(wsUrl);
            let resultText = "";

            ws.onopen = async () => {
                // 音频转base64
                const reader = new FileReader();
                reader.readAsDataURL(audioBlob);
                reader.onloadend = () => {
                    const base64Audio = reader.result.split(",")[1];
                    // 发送音频帧
                    ws.send(JSON.stringify({
                        common: { app_id: AI_CONFIG.xunfei.appId },
                        business: {
                            language: currentLanguage, // 方言配置
                            domain: "iat",
                            accent: currentLanguage === "sichuanhua" ? "mandarin" : "",
                            sample_rate: 16000,
                            encoding: "raw"
                        },
                        data: {
                            status: 2,
                            format: "audio/L16;rate=16000",
                            audio: base64Audio
                        }
                    }));
                };
            };

            ws.onmessage = (event) => {
                const data = JSON.parse(event.data);
                if (data.code !== 0) {
                    reject(new Error(data.message));
                    ws.close();
                    return;
                }
                // 拼接识别结果
                if (data.data?.result?.ws) {
                    data.data.result.ws.forEach(item => {
                        item.cw.forEach(cw => {
                            resultText += cw.w;
                        });
                    });
                }
                // 识别结束
                if (data.data.status === 2) {
                    ws.close();
                    resolve(resultText);
                }
            };

            ws.onerror = (error) => {
                reject(error);
            };

            ws.onclose = () => {
                if (resultText) resolve(resultText);
                else reject(new Error("识别失败，无返回结果"));
            };
        });
    } catch (error) {
        console.error("讯飞语音识别错误：", error);
        throw error;
    }
}

/**
 * 开始录音
 */
async function startRecording() {
    try {
        const stream = await navigator.mediaDevices.getUserMedia({ audio: { sampleRate: 16000, channelCount: 1 } });
        audioContext = new (window.AudioContext || window.webkitAudioContext)({ sampleRate: 16000 });
        const source = audioContext.createMediaStreamSource(stream);
        const processor = audioContext.createScriptProcessor(4096, 1, 1);

        audioData = [];
        processor.onaudioprocess = (e) => {
            const data = e.inputBuffer.getChannelData(0);
            audioData.push(new Int16Array(data.map(v => Math.max(-1, Math.min(1, v)) * 32767)));
        };

        source.connect(processor);
        processor.connect(audioContext.destination);
        mediaRecorder = { stream, source, processor };
        return true;
    } catch (error) {
        console.error("录音启动失败：", error);
        alert("麦克风权限被拒绝，请在浏览器地址栏左侧开启权限");
        return false;
    }
}

/**
 * 停止录音，返回音频Blob
 */
async function stopRecording() {
    return new Promise((resolve) => {
        if (!mediaRecorder) return resolve(null);

        // 停止录音
        mediaRecorder.stream.getTracks().forEach(track => track.stop());
        mediaRecorder.processor.disconnect();
        mediaRecorder.source.disconnect();
        audioContext.close();

        // 拼接音频数据，生成WAV文件
        const audioBuffer = new Int16Array(audioData.reduce((acc, val) => acc.concat(Array.from(val)), []));
        const wavBuffer = createWavFile(audioBuffer);
        const audioBlob = new Blob([wavBuffer], { type: "audio/wav" });

        // 重置变量
        audioData = [];
        mediaRecorder = null;
        audioContext = null;

        resolve(audioBlob);
    });
}

/**
 * 生成WAV格式音频文件
 */
function createWavFile(audioData) {
    const buffer = new ArrayBuffer(44 + audioData.length * 2);
    const view = new DataView(buffer);

    // WAV头文件
    writeString(view, 0, "RIFF");
    view.setUint32(4, 36 + audioData.length * 2, true);
    writeString(view, 8, "WAVE");
    writeString(view, 12, "fmt ");
    view.setUint32(16, 16, true);
    view.setUint16(20, 1, true);
    view.setUint16(22, 1, true);
    view.setUint32(24, 16000, true);
    view.setUint32(28, 16000 * 2, true);
    view.setUint16(32, 2, true);
    view.setUint16(34, 16, true);
    writeString(view, 36, "data");
    view.setUint32(40, audioData.length * 2, true);

    // 写入音频数据
    for (let i = 0; i < audioData.length; i++) {
        view.setInt16(44 + i * 2, audioData[i], true);
    }

    return buffer;
}

function writeString(view, offset, string) {
    for (let i = 0; i < string.length; i++) {
        view.setUint8(offset + i, string.charCodeAt(i));
    }
}

/**

async function callDoubaoAI(prompt) {

    if (!navigator.onLine || AI_CONFIG.openai.apiKey === "你的OpenAI API Key") {

            method: "POST",

                "Authorization": `Bearer ${AI_CONFIG.openai.apiKey}`

    }
}

/**

 */

    const inputElement = document.getElementById(inputId);
    if (!inputElement) return;
    const originText = inputElement.value;

    startSpeechRecognition(async (status, text) => {
        if (status === "onend") {
            const result = await stopSpeechRecognition();
            if (!result) return;

            // 匹配纠错指令
            const correctReg = /把(.+?)改成(.+?)/;
            const match = result.match(correctReg);
            if (match) {
                const oldText = match[1];
                const newText = match[2];
                const updatedText = originText.replace(oldText, newText);
                inputElement.value = updatedText;
                speakText(`已为您修改为${updatedText}`, currentLanguage === "sichuanhua");
            } else {
                inputElement.value = result;
                speakText(`已为您重新输入`, currentLanguage === "sichuanhua");
            }
        }
    });
};

        for (let keyword of item.keywords) {
            if (command.includes(keyword.toLowerCase())) {
                item.action();
                isMatched = true;
                break;
            }
        }
        if (isMatched) break;
    }

    // 未匹配到指令
    if (!isMatched) {
        speakText("不好意思，我没听懂，你可以说帮助，查看我能听懂的指令", currentLanguage === "sichuanhua");
    }
}

window.renderNoticeList = function() {

    if (!currentUser || !noticeModal) return;

    const notices = JSON.parse(localStorage.getItem('zhizhitong_notices')) || [];
    const userNotices = notices.filter(n => n.userId === currentUser);

        noticeList.innerHTML = '<div style="text-align: center; padding: 40px; color: #9ca3af; font-size: 14px;">暂无消息</div>';
        return;

            <div class="notice-content">${notice.content}</div>
            <div class="notice-time">${notice.time}</div>
        </div>

        item.addEventListener('click', function() {
            const noticeId = this.getAttribute('data-notice-id');
            markNoticeRead(noticeId);
        });
    });
};

        if (!newPwd) { showError('newPwd', '请设置新密码'); return; }
        if (newPwd.length < 6 || newPwd.length > 12) { showError('newPwd', '密码长度应为6-12位'); return; }
        if (newPwd !== confirmNewPwd) { showError('confirmNewPwd', '两次输入的密码不一致'); return; }

<script>
    // 表单验证
    function validateForm() {
        const phone = document.getElementById('phone').value.trim();
        const password = document.getElementById('password').value.trim();
        const confirmPassword = document.getElementById('confirmPassword').value.trim();
        let isValid = true;
        // 手机号验证
        if (!phone) { showError('phone', '请输入手机号'); isValid = false; }
        else if (!/^1[3-9]\d{9}$/.test(phone)) { showError('phone', '请输入正确的手机号'); isValid = false; }
        // 密码验证
        if (!password) { showError('password', '请输入密码'); isValid = false; }
        else if (password.length < 6 || password.length > 12) { showError('password', '密码长度应为6-12位'); isValid = false; }
        // 确认密码验证
        if (!confirmPassword) { showError('confirm', '请再次输入密码'); isValid = false; }
        else if (password !== confirmPassword) { showError('confirm', '两次输入的密码不一致'); isValid = false; }
        return isValid;
    }

    // 密码强度检测
    document.getElementById('password').addEventListener('input', function(e) {
        const password = e.target.value;
        const strength1 = document.getElementById('strength1');
        const strength2 = document.getElementById('strength2');
        const strength3 = document.getElementById('strength3');
        // 重置强度条
        strength1.className = 'strength-bar';
        strength2.className = 'strength-bar';
        strength3.className = 'strength-bar';
        // 弱：6位以上纯数字/字母
        if (password.length >= 6) {
            strength1.className = 'strength-bar weak';
        }
        // 中：8位以上，包含数字+字母
        if (password.length >= 8 && /[A-Za-z]/.test(password) && /[0-9]/.test(password)) {
            strength2.className = 'strength-bar medium';
        }
        // 强：10位以上，包含大小写+数字
        if (password.length >= 10 && /[A-Z]/.test(password) && /[a-z]/.test(password) && /[0-9]/.test(password)) {
            strength3.className = 'strength-bar strong';
        }
    });

    // 实时验证确认密码
    document.getElementById('confirmPassword').addEventListener('input', function(e) {
        const password = document.getElementById('password').value;
        const confirmPassword = e.target.value;
        const confirmError = document.getElementById('confirmError');
        if (confirmPassword && password !== confirmPassword) {
            confirmError.textContent = '两次输入的密码不一致';
            confirmError.classList.add('show');
        } else {
            confirmError.classList.remove('show');
        }
    });

    // 注册处理
    function handleRegister(btn) {
        const phone = document.getElementById('phone').value.trim();
        const password = document.getElementById('password').value.trim();
        const btnText = btn.querySelector('.btn-text');
        const loading = btn.querySelector('.loading');
        // 清除错误+验证表单
        clearErrors();
        if (!validateForm()) return;
        // 模拟加载
        btn.disabled = true;
        btnText.style.display = 'none';
        loading.classList.add('active');
        // 注册逻辑（存储用户信息）
        setTimeout(() => {
            localStorage.setItem('zhizhitong_user', JSON.stringify({ phone, pwd: password }));
            // 自动登录
            localStorage.setItem('is_login', 'true');
            localStorage.setItem('current_user', phone);
            alert('注册成功！即将跳转到首页');
            window.location.href = 'index.html';
        }, 1000);
    }
</script>
